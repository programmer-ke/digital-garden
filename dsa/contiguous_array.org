#+begin_src python

def find_max(nums: list[int]) -> int:
    max_len = 0
    csum = 0
    first_index = {0: -1}
    for i, n in enumerate(nums):
        if n == 0:
            csum += 1
        else:
            csum -= 1
        if csum in first_index:
            cur_len = i - first_index[csum]
            if cur_len > max_len:
                max_len = cur_len
        else:
            first_index[csum] = i
    return max_len

print("->", find_max([0])) # 0
print("->", find_max([0, 1])) # 2
print("->", find_max([0,1,1,1,1,1,0,0,0])) # 6

** Alternative solution

Sure. Here's a replacement of the hashmap with a fixed‑size array,
assuming you know the length =n= in advance.

*Idea* The cumulative sum ranges from =-n= to =n=.  Map a sum =s= to
index =s + n=.  Use a sentinel (e.g., =-2=) to mark that a sum hasn't
been seen yet.

*Example implementation (Python, no imports)*

#+begin_src python
def find_max_array(nums: list[int]) -> int:
    n = len(nums)
    # array of size 2*n+1, filled with sentinel -2
    first = [-2] * (2 * n + 1)
    offset = n                # to shift sum into non‑negative index
    first[offset] = -1        # sum 0 at index -1
    max_len = 0
    csum = 0
    for i, x in enumerate(nums):
        csum += 1 if x == 0 else -1   # same as before
        idx = csum + offset
        if first[idx] != -2:
            # sum seen before
            cur = i - first[idx]
            if cur > max_len:
                max_len = cur
        else:
            first[idx] = i   # store first occurrence
    return max_len

print("->", find_max_array([0])) # 0
print("->", find_max_array([0, 1])) # 2
print("->", find_max_array([0,1,1,1,1,1,0,0,0])) # 6

#+end_src
