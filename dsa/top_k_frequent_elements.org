import heapq

k = 2
counts = {1: 3, 2: 2, 3: 1}

def get_topk(counts, k):
    heap = []
    for num, freq in counts.items():
        if len(heap) == k:
            heapq.heappushpop(heap, (freq, num))
        else:
            heapq.heappush(heap, (freq, num))

#+begin_src python
def top_k_bucket(counts, k):
    # Create buckets: index = frequency (1 to max_freq)
    max_freq = max(counts.values())
    buckets = [[] for _ in range(max_freq + 1)]
    
    # Place each number in its frequency bucket
    for num, freq in counts.items():
        buckets[freq].append(num)
    
    # Collect top k from highest frequency downwards
    result = []
    for freq in range(max_freq, 0, -1):
        for num in buckets[freq]:
            result.append(num)
            if len(result) == k:
                return result
    return result

counts = {1: 3, 2: 2, 3: 1}
print(top_k_bucket(counts, 2))  # [1, 2]
