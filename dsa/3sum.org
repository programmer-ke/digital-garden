** Solution

#+begin_src python

def threesum(nums):
    nums.sort()
    i = 0
    triplets = []
    while i < len(nums) - 2:
        if nums[i] > 0:
            break
        left, right = i+1, len(nums) - 1
        while left < right:
            s = nums[i] + nums[left] + nums[right]
            if s == 0:
                triplets.append([nums[i], nums[left], nums[right]])
                left = incr(nums, left, right)
                right = decr(nums, right, left)
            elif s > 0:
                right = decr(nums, right, left)
            else:
                left = incr(nums, left, right)
        i = incr(nums, i, len(nums) - 2)
    return triplets

def incr(nums, pointer, limit):
    while pointer < limit:
        pointer += 1
        if nums[pointer] != nums[pointer - 1]:
            break
    return pointer

def decr(nums, pointer, limit):
    while pointer > limit:
        pointer -= 1
        if nums[pointer] != nums[pointer + 1]:
            break
    return pointer

#+begin_src python
def threesum(nums):
    nums.sort()
    n = len(nums)
    triplets = []
    i = 0
    while i < n - 2:
        if nums[i] > 0:
            break
        left, right = i + 1, n - 1
        while left < right:
            total = nums[i] + nums[left] + nums[right]
            if total == 0:
                triplets.append([nums[i], nums[left], nums[right]])
                left += 1
                right -= 1
                # Skip duplicates for left
                while left < right and nums[left] == nums[left - 1]:
                    left += 1
                # Skip duplicates for right
                while left < right and nums[right] == nums[right + 1]:
                    right -= 1
            elif total < 0:
                left += 1
            else:
                right -= 1
        i += 1
        # Skip duplicates for i
        while i < n - 2 and nums[i] == nums[i - 1]:
            i += 1
    return triplets

nums = [-1,0,1,2,-1,-4]                                
print(threesum(nums))
