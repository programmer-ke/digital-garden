#+begin_src python
def decoder(bytestring):
    words = []
    current_index = 0
    while current_index < len(bytestring):
        length = bytestring[current_index]
        next_word_index = current_index + length + 1
        segment = bytestring[current_index + 1:next_word_index]
        words.append(segment.decode('latin-1'))
        current_index = next_word_index
    return words

strs = ['the', 'quick', 'brown', 'fox']
print(decoder(encoder(strs)))

print(decoder(encoder([])))  # []

print(decoder(encoder(['']))) # ['']
